            <h3>?? Sistema de Busca</h3>
            <input type="text" id="inputBusca" placeholder="Buscar serviços..." onkeyup="buscarServicos()">
            <select onchange="ordenarServicos()" id="selectOrdenacao">
                <option value="nome">Ordenar por Nome</option>
                <option value="preco">Ordenar por Preço</option>
            </select>

            <h3>?? Gestão de Serviços</h3>
            <input type="text" id="inputNome" placeholder="Nome do serviço">
            <input type="text" id="inputPreco" placeholder="Preço">
            <button onclick="adicionarServico()">? Adicionar</button>

    <script>
        let servicos = [
            { id: 1, nome: "Internet Móvel", preco: "500 MT", categoria: "dados" },
            { id: 2, nome: "Voz Pré-pago", preco: "Variado", categoria: "voz" },
            { id: 3, nome: "mkesh", preco: "Grátis", categoria: "financeiro" },
            { id: 4, nome: "Banda Larga", preco: "1500 MT", categoria: "dados" }
        ];

        // BOTÃO DE RESET - NOVA FUNÇÃO
        function resetarServicos() {
            servicos = [
                { id: 1, nome: "Internet Móvel", preco: "500 MT", categoria: "dados" },
                { id: 2, nome: "Voz Pré-pago", preco: "Variado", categoria: "voz" },
                { id: 3, nome: "mkesh", preco: "Grátis", categoria: "financeiro" },
                { id: 4, nome: "Banda Larga", preco: "1500 MT", categoria: "dados" }
            ];
            salvarDados();
            atualizarLista();
            alert('? Serviços originais restaurados!');
        }

        function carregarDados() {
            const salvos = localStorage.getItem('servicosTMcel');
            if (salvos) {
                servicos = JSON.parse(salvos);
            }
            atualizarLista();
        }

        function salvarDados() {
            localStorage.setItem('servicosTMcel', JSON.stringify(servicos));
        }

        function buscarServicos() {
            const termo = document.getElementById('inputBusca').value.toLowerCase();
            if (termo === '') {
                atualizarLista();
                return;
            }
            const resultados = servicos.filter(servico => 
                servico.nome.toLowerCase().includes(termo) || 
                servico.categoria.toLowerCase().includes(termo)
            );
            exibirResultados(resultados);
        }

        function ordenarServicos() {
            const criterio = document.getElementById('selectOrdenacao').value;
            servicos.sort((a, b) => {
                if (criterio === 'nome') return a.nome.localeCompare(b.nome);
                else return a.preco.localeCompare(b.preco);
            });
            atualizarLista();
        }

        function adicionarServico() {
            const nome = document.getElementById('inputNome').value;
            const preco = document.getElementById('inputPreco').value;
            if (nome && preco) {
                servicos.push({
                    id: Date.now(),
                    nome: nome,
                    preco: preco,
                    categoria: "personalizado"
                });
                salvarDados();
                atualizarLista();
                document.getElementById('inputNome').value = '';
                document.getElementById('inputPreco').value = '';
            } else {
                alert('Preencha nome e preço!');
            }
        }

        function removerServico(id) {
            servicos = servicos.filter(servico => servico.id !== id);
            salvarDados();
            atualizarLista();
        }

        function atualizarLista() {
            const lista = document.getElementById('listaServicos');
            lista.innerHTML = '';
            servicos.forEach(servico => {
                lista.innerHTML += `
                    <div class="servico-item">
                        <div>
                            <strong>${servico.nome}</strong> - ${servico.preco}
                            <br><small>Categoria: ${servico.categoria}</small>
                        </div>
                        <button onclick="removerServico(${servico.id})">??? Remover</button>
                    </div>
                `;
            });
        }

        function exibirResultados(resultados) {
            const lista = document.getElementById('listaServicos');
            lista.innerHTML = '';
            resultados.forEach(servico => {
                lista.innerHTML += `
                    <div class="servico-item">
                        <div>
                            <strong>${servico.nome}</strong> - ${servico.preco}
                            <br><small>Categoria: ${servico.categoria}</small>
                        </div>
                        <button onclick="removerServico(${servico.id})">??? Remover</button>
                    </div>
                `;
            });
        }

        function mostrarServicos() {
            alert('Serviços TMcel:\n\n• Internet Móvel - 500MT\n• Voz Pré-pago - Variado\n• mkesh - Grátis\n• Banda Larga - 1500MT');
        }

        function mostrarSobre() {
            alert('Projeto Tecnologias Web\n\nSite TMcel Redesenhado\nTecnologias: HTML5, CSS3, JavaScript\nFuncionalidades: CRUD, Busca, Ordenação, localStorage');
        }

        document.addEventListener('DOMContentLoaded', carregarDados);
